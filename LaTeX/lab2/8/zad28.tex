\begin{tw}
Jeżeli $\lim_{n \to \infty}a_n=a$ i $\lim_{n \to \infty} b_n=\infty$
\end{tw}
\begin {enumerate}
\item $\lim_{n \to \infty}(a_n+b_n)=+\infty$
\item $ \lim_{n \to \infty}(a_n \cdot b_n)=$\left \{ \begin{array}{l}	
	$+\infty $, gdy a>0,\\
	$--\infty $, gdy a<0,
\end{array} \\$
\item $\lim_{n \to \infty}\frac{a_n}{b_n}=0$, przy $b_n \neq  0$ dla n \in N$\\
\item $\lim_{n \to \infty}\frac{a_n}{b_n}=$\left \{ \begin{array}{l} 
	$+\infty $, gdy a>0,\\
	$--\infty $, gdy a<0,
\end{array}\right} przy założeniu że $a_n \neq 0$ dla n \in N


\end {enumerate}
